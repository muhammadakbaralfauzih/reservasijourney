      $('dp_harga').value = 'Rp ' + dp.toLocaleString('id-ID'); 
      $('sisa_harga').value = 'Rp ' + (t - dp).toLocaleString('id-ID');

      const dDiv = $('paket_deskripsi_tampil');
      if(pkg && pkg.desc && pkg.desc.trim() !== '') { dDiv.innerHTML = `<span class="font-bold text-charcoal-800 mr-1"><i class="fa-solid fa-circle-info text-gold-500 mr-1.5"></i> Detail Fasilitas:</span> ${pkg.desc}`; dDiv.classList.remove('hidden'); } 
      else { dDiv.classList.add('hidden'); dDiv.innerHTML = ''; }
    }

    function getCurrentLocation() {
      if(!navigator.geolocation) return showToast('GPS Tidak didukung.', 'error');
      const b = $('btnLocation'), og = b.innerHTML; b.innerHTML = '<i class="fa-solid fa-circle-notch fa-spin text-lg"></i>';
      navigator.geolocation.getCurrentPosition(p => { $('link_maps_input').value = `https://www.google.com/maps?q=${p.coords.latitude},${p.coords.longitude}`; b.innerHTML = og; showToast('Lokasi dilacak!'); }, e => { b.innerHTML = og; showToast('Izin ditolak.', 'error'); }, {enableHighAccuracy: true});
    }
